update 修改器 ¶

一、前提 ¶

（1）添加原数据

javascript
/**
 * 构成数据（练习更新Update）
 */
var workmate1 = {
    name: 'JSPang',
    age: 33,
    sex: 1,
    job: '前端',
    skill: {
        skillOne: 'HTML+CSS',
        SkillTwo: 'JavaScript',
        SkillThree: 'PHP'
    },
    regeditTime: new Date()
}
var workmate2 = {
    name: 'ShengLei',
    age: 30,
    sex: 1,
    job: 'JAVA后端',
    skill: {
        skillOne: 'HTML+CSS',
        SkillTwo: 'J2EE',
        SkillThree: 'PPT'
    },
    regeditTime: new Date()
}
var workmate3 = {
    name: 'MinJie',
    age: 20,
    sex: 1,
    job: 'UI设计',
    skill: {
        skillOne: 'PhotoShop',
        SkillTwo: 'UI',
        SkillThree: 'Word+Excel+PPT'
    },
    regeditTime: new Date()
}
var db = connect('company');
var workmateArray = [workmate1, workmate2, workmate3];
db.workmate.insert(workmateArray);

// print('[SUCCESS]: The data was inserted successfully.');

（2）update workmate3 文件

javascript
db.workmate.update({name:'MinJie'},{age: 18})

// 结果： workmate3 = {age: 18}
// 不是想要的结果

（3） 最基础解决办法

javascript
var db=connect('company')

// 缺点：繁琐

二、$set修改器 ¶

（1）默认写法（修改） ¶

作用：用来修改一个指定的键值(key)。

JavaScript
db.workmate.update({"name":"MinJie"},{"$set":{sex:2,age:21}});

// 也可取消引号
db.workmate.update({"name":"MinJie"},{$set:{sex:2,age:21}});

// 修改后的结果
var workmate3={
    name:'MinJie',
    age:21,
    sex:2,
    job:'UI设计',
    skill:{
        skillOne:'PhotoShop',
        SkillTwo:'UI',
        SkillThree:'Word+Excel+PPT'
    },
    regeditTime:new Date()
}

（2）修改嵌套内容(内嵌文档) ¶

JavaScript
db.workmate.update({"name":"MinJie"},{"$set":{"skill.skillThree":'word'}});

// 修改后的结果
var workmate3={
    name:'MinJie',
    age:21,
    sex:2,
    job:'UI设计',
    skill:{
        skillOne:'PhotoShop',
        SkillTwo:'UI',
        SkillThree:'word'
    },
    regeditTime:new Date()
}

（3）$unset用于将key删除 ¶

作用：删除一个key值和键。

javascript
db.workmate.update({"name":"MinJie"},{$unset:{"age":''}});

// 修改后的结果
var workmate3={
    name:'MinJie',
    sex:2,
    job:'UI设计',
    skill:{
        skillOne:'PhotoShop',
        SkillTwo:'UI',
        SkillThree:'word'
    },
    regeditTime:new Date()
}

当你删除后，想加回来可以直接用set进行添加。

（4）$inc对数字进行计算 ¶

作用：它是对value值的修改，但是修改的必须是数字，字符串是不起效果的。

javascript
db.workmate.update({"name":"MinJie"},{$inc:{"age":-2}});

// 修改后的结果
var workmate3={
    name:'MinJie',
    age:18,  // 原值为20
    sex:2,
    job:'UI设计',
    skill:{
        skillOne:'PhotoShop',
        SkillTwo:'UI',
        SkillThree:'word'
    },
    regeditTime:new Date()
}

（5）multi选项 ¶

作用：可选， mongodb 默认是false,只更新找到的第一条记录，如果这个参数为true,就把按条件查出来多条记录全部更新。

javascript
// 只是更新集合（collections）第一条 【新增一个键值interset】
db.workmate.update({},{$set:{interset:[]}});

// 更新所有集合（collections） 【新增一个键值interset】
db.workmate.update({},{$set:{interset:[]}},{multi:true})

（6）upsert选项 ¶

作用：可选，这个参数的意思是，如果不存在update的记录，是否插入objNew,true为插入，默认是false，不插入。

javascript
// 如果没有更新的数据，则不插入
db.workmate.update({name:'xiaoWang'},{$set:{age:20}});

// 如果没有更新的数据，则插入该条查询的数据
db.workmate.update({name:'xiaoWang'},{$set:{age:20}},{upsert:true});

（7）$push追加数组/内嵌文档值 ¶

作用：$push的功能是追加数组中的值，但我们也经常用它操作内嵌稳文档，就是{}对象型的值。

JavaScript
// 给小王加上一个爱好(interset)为画画（draw）
db.workmate.update({name:'xiaoWang'},{$push:{interest:'draw'}});

// 给我们的MinJie，增加一项新的技能skillFour为draw
db.workmate.update({name:'MinJie'},{$push:{"skill.skillFour":'draw'}});

（8）$ne查找是否存在 ¶

作用：检查一个值是否存在，如果不存在再执行操作，存在就不执行。(总结：没有则修改，有则不修改)

javascript
// 如果xiaoWang的爱好（interest）里没有palyGame这个值，我们就加入Game这个爱好
db.workmate.update(
	{
		name:'xiaoWang',
		"interest":{$ne:'playGame'}
	},
	{
		$push:{interest:'Game'}
    }
)
$addToSet 升级版的$ne ¶
javascript
// 查看小王(xiaoWang)兴趣(interest)中有没有阅读（readBook）这项，没有则加入读书(readBook)的兴趣.
db.workmate.update({name:"xiaoWang"},{$addToSet:{interest:"readBook"}})
$each 批量追加 ¶

作用：它可以传入一个数组，一次增加多个值进去，相当于批量操作，性能同样比循环操作要好很多，这个是需要我们注意的，工作中也要先组合成数组，然后用批量的形式进行操作。

JavaScript
// 给xiaoWang,一次加入三个爱好，唱歌（Sing），跳舞（Dance），编码（Code）
var newInterset=["Sing","Dance","Code"];
db.workmate.update({name:"xiaoWang"},{$addToSet:{interest:{$each:newInterset}}})

（9）$pop 删除数组值 ¶

作用：$pop只删除一次，并不是删除所有数组中的值。而且它有两个选项，一个是1和-1。

  • 1：从数组末端进行删除
  • -1：从数组开端进行删除
JavaScript
// 现在要删除xiaoWang的编码爱好（code）
db.workmate.update({name:'xiaoWang'},{$pop:{interest:1}})

（10）数组定位修改 ¶

作用：有时候只知道修改数组的第几位，但并不知道是什么，这时候我们可以使用interest.int 的形式。

修改制定位置（int）的value值。

JavaScript
// 修改xiaoWang的第三个兴趣为编码（Code），注意这里的计数是从0开始的。
db.workmate.update({name:'xiaoWang'},{$set:{"interest.2":"Code"}})

三、修改：状态返回与安全 ¶

$set 上述操作是非应答式操作。

应答式写入 就会给我们直接返回结果（报表），结果里边的包含项会很多，这样我们就可以很好的进行程序的控制和安全机制的处理。

（1）db.runCommand( ) ¶

它是数据库运行命令的执行器，执行命令首选就要使用它，因为它在Shell和驱动程序间提供了一致的接口。（几乎操作数据库的所有操作，都可以使用runCommand来执行）

JavaScript
// 修改所有男士的数据，每个人增加了1000元钱(money)，然后用db.runCommand()执行。
db.workmate.update({sex:1},{$set:{money:1000}},false,true); 
// false：第一句末尾的false是upsert的简写，代表没有此条数据时不增加;
// true：true是multi的简写，代表修改所有，这两个我们在前边课程已经学过。

var resultMessage = db.runCommand({getLastError:1});
// getLastError:1 :表示返回功能错误，这里的参数很多。

printjson(resultMessage);
// printjson：表示以json对象的格式输出到控制台。


//  控制台返回结果
{
        "connectionId" : 101,
        "updatedExisting" : true,
        "n" : 2,
        "syncMillis" : 0,
        "writtenTo" : null,
        "err" : null,
        "ok" : 1
}

db.listCommands( ):查看所有的Commad命令

javascript
// 查看是否和数据库链接成功
db.runCommand({ping:1});

（2）findAndModify ¶

作用： findAndModify是查找并修改的意思。配置它可以在修改后给我们返回修改的结果。

findAndModify属性值：

  • query：需要查询的条件/文档
  • sort: 进行排序
  • remove：[boolean]是否删除查找到的文档，值填写true，可以删除。
  • new:[boolean]返回更新前的文档还是更新后的文档。
  • fields：需要返回的字段
  • upsert：没有这个值是否增加。
JavaScript
var myModify={
    findAndModify:"workmate",
    query:{name:'JSPang'},
    update:{$set:{age:18}},
    new:true    //更新完成，需要查看结果，如果为false不进行查看结果
}
var ResultMessage = db.runCommand(myModify);
printjson(ResultMessage);

// 控制台返回结果
{
    "lastErrorObject" : {
        "n" : 1,
        "updatedExisting" : true
    },
    "value" : {
        "_id" : ObjectId("5ca3249acaf95b982e8fbbd8"),
        "name" : "JSPang",
        "age" : 18,
        "sex" : 1,
        "job" : "前端",
        "skill" : {
             "skillOne" : "HTML+CSS",
             "SkillTwo" : "JavaScript",
             "SkillThree" : "PHP"
        },                
        "regeditTime" : ISODate("2019-04-02T09:00:10.636Z"),
        "money" : 1000        
    },
    "ok" : 1
}

findAndModify的性能是没有直接使用db.collections.update的性能好，但是在实际工作中都是使用它，毕竟要商用的程序安全性还是比较重要的。
